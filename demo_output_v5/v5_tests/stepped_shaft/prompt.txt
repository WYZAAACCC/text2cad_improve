  station1 r=25 z=0-10 (左轴肩),
  station2 r=18 z=10-50 (中段轴承位),
  station3 r=25 z=50-60 (右轴肩),
  station4 r=15 z=60-90 (右端轴颈),
  station5 r=10 z=90-100 (右端螺纹段).
