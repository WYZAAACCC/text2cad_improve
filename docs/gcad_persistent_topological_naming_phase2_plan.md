```python
class TopologyOutputRole(BaseModel):
    """声明一个 operation 产生的一种语义拓扑实体。"""

    allows_merge: bool = False
```

### 2.2 `topology/history_wrappers.py` — History-aware OCP wrappers

**设计原则**：
- 所有 OCP import 都是 lazy（函数内部 import）
- 每个函数返回 `HistoryAwareShapeResult` 或 None（OCP 不可用时）
- `capability_probe()` 在模块加载时运行，结果缓存在模块级 dict

```python
from typing import Any
from dataclasses import dataclass, field
from pydantic import BaseModel


class KernelHistorySnapshot(BaseModel):
    """Snapshot of OCCT history for one operation."""
    generated: dict[str, list[int]] = {}   # source_entity → [generated_indices]
    modified: dict[str, list[int]] = {}    # source_entity → [modified_indices]
    deleted: list[str] = []                # [deleted_entity_ids]


@dataclass
class HistoryAwareShapeResult:
    result_shape: Any
    history: KernelHistorySnapshot | None = None
    metrics: dict = field(default_factory=dict)


class KernelHistoryAdapter:
    """Lazy-import wrapper for OCCT BRepBuilderAPI_MakeShape history API."""

    def __init__(self, maker: Any):
        self._maker = maker

    def generated(self, source_shape: Any) -> list[Any]:
        try:
            return list(self._maker.Generated(source_shape))
        except Exception:
            return []

    def modified(self, source_shape: Any) -> list[Any]:
        try:
            return list(self._maker.Modified(source_shape))
        except Exception:
            return []

    def is_deleted(self, source_shape: Any) -> bool:
        try:
            return bool(self._maker.IsDeleted(source_shape))
        except Exception:
            return False
```
